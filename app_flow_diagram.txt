flowchart TD
    %% Styling
    classDef user fill:#64748b,stroke:#0f172a,stroke-width:2px,color:#fff
    classDef auth fill:#f97316,stroke:#c2410c,stroke-width:2px,color:#fff
    classDef main fill:#3b82f6,stroke:#1e3a8a,stroke-width:2px,color:#fff
    classDef action fill:#10b981,stroke:#064e3b,stroke-width:2px,color:#fff
    classDef ai fill:#8b5cf6,stroke:#4c1d95,stroke-width:2px,color:#fff
    classDef decision fill:#eab308,stroke:#a16207,stroke-width:2px,color:#fff

    User(["User (Admin or Member)"]):::user
    
    %% Authentication Flow
    Login["Login / Register"]:::auth
    Dash["Dashboard (View Workspaces)"]:::main
    
    User --> Login
    Login --> Dash

    %% Main Application Flow
    Dash -->|Selects| WS["Workspace View"]:::main
    
    %% Branching from Workspace
    WS -->|Action| CreateTicket["Create New Ticket"]:::action
    WS -->|Action| ManageBoard["Drag & Drop Kanban Board"]:::action
    WS -->|Action| ManageMeeting["Start / Join Meeting"]:::action
    
    %% Ticket Lifecycle
    CreateTicket --> TicketCreated["Ticket in 'Todo'"]:::action
    ManageBoard -->|Update Status| TicketMove["Move to In Progress / Done"]:::action
    
    %% AI Features
    TicketCreated -.->|Triggers| AIPriority{"AI Priority Agent"}:::ai
    AIPriority -.->|Analyzes & Sets| AutoPriority["Auto-Assigns Priority (Low/Med/High)"]:::action
    
    %% Meeting & Transcription Flow
    ManageMeeting --> RecordSpeech["Record Speech-to-Text"]:::action
    RecordSpeech --> StopMeeting["End Meeting"]:::action
    StopMeeting -.->|Triggers| AINotes{"AI Summarization Agent"}:::ai
    AINotes -.->|Generates| SaveNotes["Save Transcripts & Summary"]:::action
    
    %% Admin Specific Flow
    Dash -->|If Admin| AdminPanel["Admin Panel"]:::main
    AdminPanel --> Invite["Invite Members to Workspace"]:::action
    AdminPanel --> ViewAll["View All Users/Metrics"]:::action
    
    %% GitHub Integration Flow (Automated)
    GitHub["GitHub (Developer Commits/PRs)"]:::decision
    GitHub -.->|Webhook Event| GHHook["GitHub Webhook Processor"]:::action
    GHHook -.->|Auto Updates| AutoTicketMove["Update Ticket Status (e.g. to Done)"]:::action

    %% Link back to main views
    TicketMove --> WS
    AutoPriority --> WS
    SaveNotes --> WS
    Invite --> WS
    AutoTicketMove --> WS
